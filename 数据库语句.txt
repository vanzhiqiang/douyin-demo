CREATE TABLE `user` (
`Id` int(10) NOT NULL AUTO_INCREMENT,
`Name` varchar(20) DEFAULT NULL,
`Password` varchar(20) DEFAULT NULL,
`FollowCount` int(20) DEFAULT NULL,
`FollowerCount` int(20) DEFAULT NULL,
PRIMARY KEY (`Id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;


CREATE TABLE `video` (
`Id` int(10) NOT NULL AUTO_INCREMENT,
`UserId` int(10) DEFAULT NULL,
`PlayUrl` varchar(50) DEFAULT NULL,
`CoverUrl` varchar(200) DEFAULT NULL,
`FavoriteCount` int(20) DEFAULT NULL,
`CommentCount` int(20) DEFAULT NULL,
`IsFavorite` boolean DEFAULT NULL,
PRIMARY KEY (`Id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;


CREATE TABLE `favorite` (
`Id` int(10) NOT NULL AUTO_INCREMENT,
`UserId` int(10) DEFAULT NULL,
`VideoId` int(10) DEFAULT NULL,
`IsFavorite` boolean DEFAULT NULL,
PRIMARY KEY (`Id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;


CREATE TABLE `relation` (
`Id` int(10) NOT NULL AUTO_INCREMENT,
`FollowId` int(10) DEFAULT NULL,
`FollowerId` int(10) DEFAULT NULL,
`IsFollow` boolean DEFAULT NULL,
PRIMARY KEY (`Id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;


CREATE TABLE `comment` (
`Id` int(10) NOT NULL AUTO_INCREMENT,
`VideoId` int(10) DEFAULT NULL,
`UserId` int(10) DEFAULT NULL,
`Content` varchar(200) DEFAULT NULL,
`CreateDate` varchar(50) DEFAULT NULL,
PRIMARY KEY (`Id`)
) ENGINE=InnoDB AUTO_INCREMENT=1 DEFAULT CHARSET=utf8;
